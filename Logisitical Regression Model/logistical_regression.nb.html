lr_auc <- function(train, test) {
  lr_model <- glm(loan_status ~ annual_inc + dti + fico_range_low + term + int_rate, data = train, family = binomial)
  pred <- predict(lr_model, test, type = "response")
  roc(test$loan_status, pred)$auc
}
